Creation of Replicaset and Deployment 

 mkdir yaml
cd yaml/
      vim replica.yaml
apiVersion: apps/v1
kind: ReplicaSet
metadata:
  name: my-replicaset
spec:
  replicas: 2
  selector:
    matchLabels:
      app: my-app
  template:
    metadata:
      labels:
        app: my-app
    spec:
      containers:
      - name: my-container
        image: nginx
 
kubectl get node  .......check node
    kubectl get pod  .....check pod
     kubectl api-resources      .....checking api-resources
     kubectl api-resources | grep replicaset      ......checking api-resources for replicaset ....<apps>
     kubectl api-resources | grep pod           ..........checking api-resources for pod
     kubectl apply -f replica.yaml      .....<for updating your yaml file>
     kubectl get rs          ........checking no of replica created
    kubectl get pod      .....checking no of pods
     cp replica.yaml deployment.yaml      .....copying replica.yaml file content to deployment.yaml
     cat deployment.yaml
  -----   vim deployment.yaml      ....note: it's kind must be-Deployment
apiVersion: apps/v1
kind: Deployment
metadata:
  name: my-replicaset
spec:
  replicas: 2
  selector:
    matchLabels:
      app: my-app
  template:
    metadata:
      labels:
        app: my-app
    spec:
      containers:
      - name: my-container
        image: nginx
     
kubectl apply -f deployment.yaml
    kubectl get deployment
    kubectl get all
